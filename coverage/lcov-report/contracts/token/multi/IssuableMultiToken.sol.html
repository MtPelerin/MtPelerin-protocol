pragma solidity ^0.4.24;
 
import "../../zeppelin/math/SafeMath.sol";
import "./BasicMultiToken.sol";
 
 
/**
 * @title IssuableMultiToken
 * @dev BasicMultiToken contract which implement an issuing mechanism.
 * @author Cyril Lapinte - <cyril.lapinte@mtpelerin.com>
 */
contract IssuableMultiToken is BasicMultiToken {
  using SafeMath for uint256;
 
  /* allTimeIssued_ and allTimeRedeemed_ 
   * are for information only
   * as overflow may happend in a distant future */
  mapping(uint256 => uint256) internal allTimeIssued_;
  mapping(uint256 => uint256) internal allTimeRedeemed_;
 
  /**
   * @dev all time issued for a currency
   */
  function allTimeIssued(uint256 _currencyId) public view returns (uint256) {
    return allTimeIssued_[_currencyId];
  }
 
  /**
   * @dev all time redeemed for a currency
   */
  function allTimeRedeemed(uint256 _currencyId) public view returns (uint256) {
    return allTimeRedeemed_[_currencyId];
  }
 
  /**
   * @dev called by the owner to increase the supply
   */
  function issue(uint256 _currencyId, uint256 _amount) onlyOwner public {
    balances[_currencyId][owner] = balances[_currencyId][owner].add(_amount);
    totalSupplies_[_currencyId] = totalSupplies_[_currencyId].add(_amount);
 
    allTimeIssued_[_currencyId] += _amount;
    emit Issued(_currencyId, _amount);
  }
 
  /**
   * @dev called by the owner to decrease the supply
   */
  function redeem(uint256 _currencyId, uint256 _amount) onlyOwner public {
    balances[_currencyId][owner] = balances[_currencyId][owner].sub(_amount);
    totalSupplies_[_currencyId] = totalSupplies_[_currencyId].sub(_amount);
 
    allTimeRedeemed_[_currencyId] += _amount;
    emit Redeemed(_currencyId, _amount);
  }
 
  event Issued(uint256 currencyId, uint256 amount);
  event Redeemed(uint256 currencyId, uint256 amount);
}
