pragma solidity ^0.4.24;
 
import "../../zeppelin/math/SafeMath.sol";
import "./BasicMultiToken.sol";
 
 
/**
 * @title SeizableMultiToken
 * @dev BasicToken contract which allows owner to seize accounts
 * @author Cyril Lapinte - <cyril.lapinte@mtpelerin.com>
 */
contract SeizableMultiToken is BasicMultiToken {
  using SafeMath for uint256;
 
  mapping(uint256 => uint256) internal allTimeSeized_;
 
  /**
   * @dev all time seized for a currency
   */
  function allTimeSeized(uint256 _currencyId) public view returns (uint256) {
    return allTimeSeized_[_currencyId];
  }
 
  /**
   * @dev called by the owner to seize value from the account
   */
  function seize(uint256 _currencyId, address _account, uint256 _value)
    onlyOwner public
  {
    require(_account != owner);
 
    balances[_currencyId][_account] = balances[
      _currencyId][_account].sub(_value);
    balances[_currencyId][owner] = balances[_currencyId][owner].add(_value);
 
    allTimeSeized_[_currencyId] += _value;
    emit Seized(_currencyId, _account, _value);
  }
 
  event Seized(uint256 currencyId, address account, uint256 amount);
 
}
